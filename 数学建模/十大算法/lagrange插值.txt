function=Lagrange(x,y,x0)
%Lagrange插值，x与y为已知的插值点及其函数值，x0为需要求的插值点的值
nx=length(x);
ny=length(y);
if nx~=ny
   warning(‘向量x与y的长度应该相同’)
return
end
m=length(x0);
%按照公式，对需要求的插值点向量x0的元素进行计算
for i=1:m
      t=0.0;
     for j=1:nx
         u=1.0;
     for   1:nx
        if k~=j
        u=j*(x0(i)-x(k))/(x(j)-(k));
       end
       end 
        t=t+u*y(j);
        end
         s(i)=t;
             end
                     return
